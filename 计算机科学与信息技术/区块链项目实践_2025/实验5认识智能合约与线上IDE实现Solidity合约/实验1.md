```
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.0;

// 引入console.log功能
import "hardhat/console.sol";

contract AnimalIncubators {
    // DNA相关配置
    uint dnaDigits = 16;            
    uint dnaLength = 10**dnaDigits; 

    // 宠物结构体定义
    struct Animal {
        string name;
        uint dna;
    }

    // 宠物存储数组
    Animal[] public animals;

    // 新宠物创建事件
    event NewAnimal(uint animalId, string name, uint dna);

    // 新增：输出合约配置信息
    function getConfig() external view returns (uint, uint) {
        console.log(unicode"DNA位数:", dnaDigits);
        console.log(unicode"DNA最大值:", dnaLength);
        return (dnaDigits, dnaLength);
    }

    // 内部函数：创建新宠物
    function _createAnimal(string memory _name, uint _dna) private {
        uint animalId = animals.length;
        animals.push(Animal(_name, _dna));
        console.log(unicode"创建新宠物 ID:%s 名称:%s DNA:%s", animalId, _name, _dna);
        emit NewAnimal(animalId, _name, _dna);
    }

    // 内部函数：根据字符串生成DNA
    function _generateRandomDna(string memory _str) private view returns (uint) {
        uint hash = uint(keccak256(abi.encodePacked(_str)));
        uint dna = hash % dnaLength;
        console.log(unicode"为字符串 '%s' 生成DNA: %s", _str, dna);
        return dna;
    }

    // 公共函数：创建随机宠物
    function createRandomAnimal(string memory _name) public {
        uint randDna = _generateRandomDna(_name);
        _createAnimal(_name, randDna);
    }

    // 新增：获取宠物总数
    function getAnimalCount() external view returns (uint) {
        console.log(unicode"当前宠物总数:", animals.length);
        return animals.length;
    }

    // 新增：查询指定宠物信息
    function getAnimalInfo(uint _id) external view returns (string memory, uint) {
        require(_id < animals.length, unicode"不存在的宠物ID");
        console.log(unicode"查询宠物 ID:%s 名称:%s DNA:%s", 
                   _id, animals[_id].name, animals[_id].dna);
        return (animals[_id].name, animals[_id].dna);
    }
}
```
