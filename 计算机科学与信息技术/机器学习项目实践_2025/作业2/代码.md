**车牌定位完整代码**
```python
import cv2
import numpy as np
from sklearn.cluster import KMeans

def locate_license_plate(img_path, k=3):
    # 1. 图像读取与预处理
    img = cv2.imread(img_path)
    img_resized = cv2.resize(img, (600, 400))  # 调整大小加速处理
    hsv = cv2.cvtColor(img_resized, cv2.COLOR_BGR2HSV)  # 转HSV颜色空间

    # 2. K-means颜色聚类分割
    pixels = hsv.reshape(-1, 3)
    kmeans = KMeans(n_clusters=k, n_init=10)
    labels = kmeans.fit_predict(pixels)
    clustered = kmeans.cluster_centers_[labels].reshape(hsv.shape).astype(np.uint8)

    # 3. 提取目标颜色区域（假设车牌为蓝色或黄色）
    mask = cv2.inRange(clustered, 
                      (100, 50, 50),   # 蓝色HSV下限
                      (140, 255, 255)) # 蓝色HSV上限
    # 或提取黄色：
    # mask = cv2.inRange(clustered, (20, 100, 100), (30, 255, 255))

    # 4. 形态学操作连接区域
    kernel = cv2.getStructuringElement(cv2.MORPH_RECT, (5, 5))
    closed = cv2.morphologyEx(mask, cv2.MORPH_CLOSE, kernel, iterations=3)

    # 5. 查找轮廓并筛选车牌候选区域
    contours, _ = cv2.findContours(closed, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE)
    candidates = []
    for cnt in contours:
        x, y, w, h = cv2.boundingRect(cnt)
        aspect_ratio = w / h
        area = cv2.contourArea(cnt)
        # 根据长宽比和区域面积筛选
        if 2.5 < aspect_ratio < 5 and area > 1000:
            candidates.append((x, y, w, h))
    
    # 6. 在原图上绘制车牌区域
    if candidates:
        best = max(candidates, key=lambda c: c[2]*c[3])  # 选择最大区域
        x, y, w, h = best
        cv2.rectangle(img_resized, (x, y), (x+w, y+h), (0, 255, 0), 2)
    return img_resized

# 调用函数并显示结果
result = locate_license_plate("car.jpg", k=4)
cv2.imshow("Result", result)
cv2.waitKey(0)
cv2.destroyAllWindows()
```

---

**代码解析**
**1. 颜色聚类（关键逻辑）**
• HSV颜色空间：相较于RGB，HSV对光照变化更鲁棒，蓝色车牌对应HSV范围为 `H≈120°, S>50, V>50`。

• K-means参数：`k=3` 表示分割背景、车牌颜色和文字颜色，可调整 `k` 的数值适应复杂场景。


**2. 形态学操作**
• 闭运算（Closing）：填充车牌内部文字间隙，使区域连通（`kernel`大小需根据图像分辨率调整）。


**3. 轮廓筛选**
• 长宽比过滤：国内车牌长宽比约为 3.14（440mm×140mm），故设定为 2.5~5。

• 面积过滤：消除小面积噪声干扰（`area > 1000` 需根据图像分辨率调整）。


---

**定位策略改进建议**
1. 颜色动态检测：
   • 蓝色、黄色、绿色车牌需自动识别，示例代码添加以下判断逻辑：

   ```python
   # 统计主要颜色聚类中心，自适应选择
   dominant_color = kmeans.cluster_centers_[np.argmax(np.bincount(labels))]
   h_value = dominant_color[0]
   if 100 < h_value < 140:  # 蓝色
       lower, upper = (100, 50, 50), (140, 255, 255)
   elif 20 < h_value < 30:   # 黄色
       lower, upper = (20, 100, 100), (30, 255, 255)
   ```

2. 边缘密度增强：
   • 车牌区域文字边缘密集，可结合边缘检测（Canny）与颜色分割：

   ```python
   edges = cv2.Canny(img_resized, 100, 200)
   combined = cv2.bitwise_and(closed, closed, mask=edges)
   ```
